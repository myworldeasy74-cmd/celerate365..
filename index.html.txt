  // Helper: get today's date object
  function getToday() {
    const now = new Date();
    return { month: now.getMonth() + 1, day: now.getDate(), year: now.getFullYear() };
  }

  // Get celebrations for a specific month/day
  function getCelebrationsByDate(month, day) {
    return celebrationsDB.filter(celeb => celeb.month === month && celeb.day === day);
  }

  // Display today's events with animations
  function renderTodayCelebrations() {

    const celebrations = getCelebrationsByDate(today.month, today.day);

  // ---------- CALENDAR MODULE (basic interactive) ----------
  function buildCalendar(year, month) {
    const firstDay = new Date(year, month, 1).getDay(); // 0 sun
    const daysInMonth = new Date(year, month + 1, 0).getDate();
    const todayObj = getToday();
    const todayYear = todayObj.year;
    const todayMonth = todayObj.month - 1;
    const todayDateNum = todayObj.day;

    let calendarDaysHtml = '';
    for (let i = 0; i < firstDay; i++) {
      calendarDaysHtml += `<div class="cal-day empty"></div>`;
    }
    for (let d = 1; d <= daysInMonth; d++) {
      const isToday = (year === todayYear && month === todayMonth && d === todayDateNum);
      const currentMonthNum = month + 1;
      calendarDaysHtml += `<div class="cal-day ${isToday ? 'today' : ''}" data-day="${d}" data-month="${currentMonthNum}" data-year="${year}">${d}</div>`;
    }
    return calendarDaysHtml;
  }
